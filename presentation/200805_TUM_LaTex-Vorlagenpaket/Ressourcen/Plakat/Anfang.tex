% Fusszeile:

\newcommand{\PlakatFusszeile}[1]{
    \fancyfoot[L]{%
        \begin{textblock*}{\textwidth}[0,0](\PlakatPositionLinks,
                    \paperheight - \PlakatSeiteRand - \PlakatFusszeileHoehe - \PlakatFusszeilePositionUnten)%
            \begin{minipage}[b][\PlakatFusszeileHoehe][b]{\textwidth}%
                \PlakatFusszeileSchrift%
                #1%
            \end{minipage}%
        \end{textblock*}%
    }
}

\newcommand{\PlakatFusszeileMehrspaltig}[1]{
    \PlakatFusszeile{%
        \setlength{\columnsep}{\PlakatSpaltenAbstand}%
        \begin{multicols}{\PlakatFusszeileSpaltenzahl}%
            #1%
        \end{multicols}%
    }
}
